An example of a custom skin

```
skinparam monochrome false
skinparam shadowing true
skinparam backgroundColor #ffffff

skinparam defaultFontColor #333333
skinparam defaultFontName Helvetica

skinparam ParticipantPadding 20
skinparam BoxPadding 10

skinparam DatabaseBorderColor #000000
skinparam DatabaseBackgroundColor #ffffff

skinparam ControlBorderColor #000000
skinparam ControlBackgroundColor #ffffff

skinparam CollectionsBorderColor #000000
skinparam CollectionsBackgroundColor #ffffff

skinparam NoteBackgroundColor #ffee00
skinparam NoteBorderColor #000000

skinparam sequence {
    ArrowColor #000000
    LifeLineBorderColor #000000
    LifeLineBackgroundColor #cccccc
    
    BoxBorderColor none
    BoxBackgroundColor #EFF8FB

    DividerBackgroundColor #ffee00
    DividerBorderColor #666666
    DividerBorderThickness 1px

    GroupBackgroundColor #ffee00
    GroupBodyBackgroundColor none
    GroupBorderColor #333333
    GroupBorderThickness 1

    ActorBorderColor #000000
    ActorBackgroundColor #ffffff
    
    ParticipantBorderColor #000000
    ParticipantBackgroundColor #ffffff
}
```
